SELECT * FROM 
TTB01_TBL_CT CT
INNER JOIN 
TTB01_TBL_TDT TDT ON (TDT.TRAIN_SRC = CT.CITY_ID)
INNER JOIN 
TTB01_TBL_TDT TDTI ON (TDTI.TRAIN_DEST = CT.CITY_ID)
INNER JOIN
TTB01_TBL_DTB DTB ON (TDT.TRAIN_ID = DTB.TRAIN_ID);


SELECT TRAIN_NO,TRAIN_NAME,TRAIN_CITIES,TIME_TAKEN,TRAIN_SRC,TRAIN_DEST,TRAVEL_DATE,AVAILABLE_DAY,AVAILABLE_SEATS,FARE FROM
TTB01_TBL_TDT TDT
INNER JOIN
TTB01_TBL_CT CT ON (CT.CITY_ID = TDT.TRAIN_SRC) 
INNER JOIN
TTB01_TBL_CT CTI ON (CTI.CITY_ID = TDT.TRAIN_DEST)
INNER JOIN
TTB01_TBL_DTB DTB ON (DTB.TRAIN_ID = TDT.TRAIN_ID)
WHERE (CT.CITY_NAME = 'ABC' AND CTI.CITY_NAME = 'PQR' AND TRAVEL_DATE = '18-JUN-19');

TRAIN_NO,TRAIN_NAME,TRAIN_CITIES,TIME_TAKEN,TRAIN_SRC,TRAIN_DEST,TRAIN_DATE,AVAILABLE_DAY,AVAILABLE_SEATS,FARE
